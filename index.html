class DataPipeline {
    constructor(data) {
        this.data = data;
    }

    normalize() {
        const max = Math.max(...this.data);
        const min = Math.min(...this.data);

        return this.data.map(value => (value - min) / (max - min));
    }

    filter(threshold) {
        return this.data.filter(value => value > threshold);
    }

    summary() {
        const total = this.data.reduce((sum, v) => sum + v, 0);
        const mean = total / this.data.length;

        const variance = this.data.reduce(
            (acc, v) => acc + Math.pow(v - mean, 2),
            0
        ) / this.data.length;

        return {
            count: this.data.length,
            mean: mean.toFixed(2),
            variance: variance.toFixed(2),
        };
    }
}


const rawData = [12, 18, 25, 32, 45, 50, 62, 78];

const pipeline = new DataPipeline(rawData);

const normalizedData = pipeline.normalize();


const filteredData = pipeline.filter(0.4);


const stats = pipeline.summary();

console.log("Normalized Data:", normalizedData);
console.log("Filtered Data:", filteredData);
console.log("Statistics Summary:", stats);
